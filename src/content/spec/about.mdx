<script is:inline>{`
  (() => {
    const initAboutMoreAnimations = () => {
      document.querySelectorAll(".about-more").forEach((details) => {
        if (details.dataset.aboutMoreBound === "true") return;

        const summary = details.querySelector("summary");
        const body = details.querySelector(".about-more-body");
        if (!summary || !body) return;

        details.dataset.aboutMoreBound = "true";

        summary.addEventListener("click", (event) => {
          event.preventDefault();

          const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)").matches;
          if (prefersReducedMotion) {
            details.open = !details.open;
            return;
          }

          body.style.overflow = "hidden";

          if (details.open) {
            body.style.height = body.scrollHeight + "px";
            body.style.opacity = "1";

            requestAnimationFrame(() => {
              body.style.height = "0px";
              body.style.opacity = "0";
            });

            const onCloseEnd = (transitionEvent) => {
              if (transitionEvent.propertyName !== "height") return;
              body.removeEventListener("transitionend", onCloseEnd);
              details.open = false;
              body.style.height = "";
              body.style.opacity = "";
              body.style.overflow = "";
            };

            body.addEventListener("transitionend", onCloseEnd);
            return;
          }

          details.open = true;
          body.style.height = "0px";
          body.style.opacity = "0";

          requestAnimationFrame(() => {
            body.style.height = body.scrollHeight + "px";
            body.style.opacity = "1";
          });

          const onOpenEnd = (transitionEvent) => {
            if (transitionEvent.propertyName !== "height") return;
            body.removeEventListener("transitionend", onOpenEnd);
            body.style.height = "auto";
            body.style.opacity = "";
            body.style.overflow = "";
          };

          body.addEventListener("transitionend", onOpenEnd);
        });
      });
    };

    initAboutMoreAnimations();
    document.addEventListener("astro:page-load", initAboutMoreAnimations);
    document.addEventListener("swup:contentReplaced", initAboutMoreAnimations);
  })();
`}</script>
